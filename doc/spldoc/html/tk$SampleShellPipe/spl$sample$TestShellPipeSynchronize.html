 composite TestShellPipeInputSynchronize {
 
     type 
 
     InputType =
         float64 stdinTimestamp,
         int64 stdinNumber,
         rstring stdinLine;
 
     OutputType =
         float64 stdinTimestamp,
         int64 stdinNumber,
         rstring stdinLine,
         rstring stdoutLine;
 
     graph
 
 
     // create a stream of tuples containing text lines for the ShellPipe operators below to consume
     stream<InputType> InputStream as Out = FileSource() {
     param
         file: dataDirectory() + "/ozymandias.txt";
         format: line;
    output Out:
         stdinTimestamp = getTimestampInSecs(),
         stdinNumber = TupleNumber(); }
     () as DebugInputStream = FileSink(InputStream) { param file: "debug.TestShellPipeInputSynchronize.InputStream.out"; format: txt; hasDelayField: true; flush: 1u; writePunctuations: true; }
 
 
     // execute an inline Perl script that copies lines from STDIN to STDOUT
     stream<OutputType> UnsynchronizedOutputStream = ShellPipe(InputStream) {
     logic state: { mutable int64 counter = 0l; }
     param
         command: "perl -n -e \"print STDOUT \\\"unsync'ed line: \\$_\\\";\""; 
         stdinAttribute: stdinLine;
         stdoutAttribute: "stdoutLine"; }
     () as DebugUnsynchronizedOutputStream = FileSink(UnsynchronizedOutputStream) { param file: "debug.TestShellPipeInputSynchronize.UnsynchronizedOutputStream.out"; format: txt; hasDelayField: true; flush: 1u; writePunctuations: true; }
 
 
     // execute an inline Perl script that copies lines from STDIN to STDOUT
     stream<rstring stdoutLine> ShellPipeLineOutputStream as Out = ShellPipe(InputStream) {
     param
         command: "perl -n -e \"print STDOUT \\\"sync'ed line: \\$_\\\";\""; }
     () as DebugShellPipeLineOutputStream = FileSink(ShellPipeLineOutputStream) { param file: "debug.TestShellPipeInputSynchronize.ShellPipeLineOutputStream.out"; format: txt; hasDelayField: true; flush: 1u; writePunctuations: true; }
 
 
     // pair each tuple consumed by the ShellPipe operator with a tuple produced by it
     stream<OutputType> SynchronizedLineOutputStream = Barrier(InputStream ; ShellPipeLineOutputStream) {}
     () as DebugSynchronizedLineOutputStream = FileSink(SynchronizedLineOutputStream) { param file: "debug.TestShellPipeInputSynchronize.SynchronizedLineOutputStream.out"; format: txt; hasDelayField: true; flush: 1u; writePunctuations: true; }
 
 }
